import turtle

def draw_square():  #draw square
    if __name__ == '__main__':      # if statement to run the code
        window = turtle.Screen()
        window.bgcolor("red")
        brad = turtle.Turtle()
        brad.shape("turtle")
        brad.color("yellow")
        brad.speed(2)
        for i in range(4):      #for loop to draw square
            brad.forward(100)
            brad.right(90)
        window.exitonclick()
        
def draw_circle():          #draw circle
    if __name__ == '__main__':
